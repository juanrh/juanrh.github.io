00312     private void processData() {            
00313         if (m_stream0.hasMoreInstances()) {
00314             timestamp++;
00315             nProcessed++;
00316 
00317             if (timestamp % 100 == 0) 
00318                 m_visualPanel.setProcessedPointsCounter(timestamp);
00319 
00320             Instance nextStreamObj0 = m_stream0.nextInstance();
00321             DataPoint point0 = new DataPoint(nextStreamObj0,timestamp);
00322 
00323             pointBuffer0.add(point0);
00324             while (pointBuffer0.size() > m_stream0_decayHorizon) {
00325                 pointBuffer0.removeFirst();
00326             }
00327             
00328             MeasuredProcessStreamObj(ALGORITHM_1, nextStreamObj0);
00329             if (bUseAlgorithm2)
00330                 MeasuredProcessStreamObj(ALGORITHM_2, nextStreamObj0);
00331             
00332             // draw new point
00333             m_streampanel[ALGORITHM_1].drawPoint(point0, false, false);
00334             if (bUseAlgorithm2)
00335                 m_streampanel[ALGORITHM_2].drawPoint(point0, false, false);
00336             
00337             // apply decay at points    
00338             if (nProcessed % m_redrawInterval == 0) {                
00339                 float f;
00340                 if (m_stream0_decayHorizon <= m_redrawInterval) 
00341                     f = 1;
00342                 else
00343                     f = ((float)m_redrawInterval) / ((float)m_stream0_decayHorizon);
00344                 m_streampanel[ALGORITHM_1].applyDrawDecay(f, false);
00345                 if (bUseAlgorithm2)
00346                     m_streampanel[ALGORITHM_2].applyDrawDecay(f, false);
00347             }
00348             
00349             // draw events
00350             //if (nProcessed % m_eventsInterval == 0) {
00351             drawEvents();
00352             
00353             // redraw point layer
00354             m_streampanel[ALGORITHM_1].RedrawPointLayer();
00355             if (bUseAlgorithm2)
00356                 m_streampanel[ALGORITHM_2].RedrawPointLayer();
00357             
00358             /*if (nProcessed % m_drawOutliersInterval == 0)         
00359                 drawOutliers();*/
00360 
00361             if (CanPause()) {
00362                 // automatically pause each m_pauseInterval objects
00363                 
00364                 //updatePointsWeight();
00365                 showOutliers(true);
00366                 drawOutliers();
00367 
00368                 m_visualPanel.toggleVisualizer(true);
00369             }
00370         } else {
00371             System.out.println("DONE");
00372             pause();
00373             return;
00374         }
00375 
00376         simulateVisualSpeed(); // simulate visualization speed 
00377     }
