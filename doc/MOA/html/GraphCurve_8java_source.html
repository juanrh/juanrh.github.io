00079     private void paintFullCurve(Graphics g, MeasureCollection m, int mSelect, Color color){
00080             if (m.getNumberOfValues(mSelect)==0) return;
00081 
00082             boolean corrupted = false;

00088             int[] x = new int[n];

00091             for (int i = 0; i < n; i ++) {
00092                 if(x_resolution > 1){
00093                     //we need to compress the values
00094                     double sum_y = 0;

00096                     for (int j = 0; j < x_resolution; j++) {
00097                         if((i)*x_resolution+j<m.getNumberOfValues(mSelect)){
00098                             sum_y+= m.getValue(mSelect,i);
00099                             counter++;
00100                         }
00101                         sum_y/=counter;
00102                     }
00103                     x[i] = i;
00104                     y[i] = (int)(height-(sum_y/max_value)*height);
00105                 }
00106                 else{
00107                     //spreading one value
00108                     x[i] = (i)*(int)(1/x_resolution)+(int)(1/x_resolution/2);
00109                     double value = m.getValue(mSelect,i);
00110                     if(Double.isNaN(value)){
00111                         corrupted = true;
00112                         break;
00113                     }
00114                     y[i] = (int)(height-(value/max_value)*height);
00115                     
00116                 }
00117             }
00118             if(!corrupted){
00119                 g.setColor(color);
