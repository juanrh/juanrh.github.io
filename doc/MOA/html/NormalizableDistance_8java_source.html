00396   protected double difference(int index, double val1, double val2) {
00397     switch (m_Data.attribute(index).type()) {
00398       case Attribute.NOMINAL:
00399         if (isMissingValue(val1) ||
00400            isMissingValue(val2) ||
00401            ((int) val1 != (int) val2)) {
00402           return 1;
00403         }
00404         else {
00405           return 0;
00406         }
00407         
00408       case Attribute.NUMERIC:
00409         if (isMissingValue(val1) ||
00410            isMissingValue(val2)) {
00411           if (isMissingValue(val1) &&
00412              isMissingValue(val2)) {
00413             if (!m_DontNormalize)  //We are doing normalization
00414               return 1;
00415             else
00416               return (m_Ranges[index][R_MAX] - m_Ranges[index][R_MIN]);
00417           }
00418           else {
00419             double diff;
00420             if (isMissingValue(val2)) {
00421               diff = (!m_DontNormalize) ? norm(val1, index) : val1;
00422             }
00423             else {
00424               diff = (!m_DontNormalize) ? norm(val2, index) : val2;
00425             }
00426             if (!m_DontNormalize && diff < 0.5) {
00427               diff = 1.0 - diff;
00428             }
00429             else if (m_DontNormalize) {
00430               if ((m_Ranges[index][R_MAX]-diff) > (diff-m_Ranges[index][R_MIN]))
00431                 return m_Ranges[index][R_MAX]-diff;
00432               else
00433                 return diff-m_Ranges[index][R_MIN];
00434             }
00435             return diff;
00436           }
00437         }
00438         else {
00439           return (!m_DontNormalize) ? 
00440                  (norm(val1, index) - norm(val2, index)) :
00441                  (val1 - val2);
00442         }
00443         
00444       default:
00445         return 0;
00446     }
