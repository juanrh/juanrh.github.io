00113         double d = 1.0;
00114         int[] m = new int[this.ensemble.length];
00115         for (int j = 0; j < this.ensemble.length; j++) {
00116             int j0 = 0; //max(0,j-K)
00117             pipos[j] = 1.0;
00118             pineg[j] = 1.0;
00119             m[j] = -1;
00120             if (this.ensemble[j].correctlyClassifies(inst) == true) {
00121                 m[j] = 1;
00122             }
00123             for (int k = j0; k <= j - 1; k++) {
00124                 pipos[j] *= wpos[j][k] / wpos[j][j] * Math.exp(-alphainc[k])
00125                         + (1.0 - wpos[j][k] / wpos[j][j]) * Math.exp(alphainc[k]);
00126                 pineg[j] *= wneg[j][k] / wneg[j][j] * Math.exp(-alphainc[k])
00127                         + (1.0 - wneg[j][k] / wneg[j][j]) * Math.exp(alphainc[k]);
00128             }
00129             for (int k = 0; k <= j; k++) {
00130                 wpos[j][k] = wpos[j][k] * pipos[j] + d * (m[k] == 1 ? 1 : 0) * (m[j] == 1 ? 1 : 0);
00131                 wneg[j][k] = wneg[j][k] * pineg[j] + d * (m[k] == -1 ? 1 : 0) * (m[j] == -1 ? 1 : 0);
00132             }
00133             alphainc[j] = -alpha[j];
00134             alpha[j] = 0.5 * Math.log(wpos[j][j] / wneg[j][j]);
00135             alphainc[j] += alpha[j];

00137             d = d * Math.exp(-alpha[j] * m[j]);
