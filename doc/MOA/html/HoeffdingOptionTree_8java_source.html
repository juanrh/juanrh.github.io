00955     public void enforceTrackerLimit() {
00956         if ((this.inactiveLeafNodeCount > 0)
00957                 || ((this.activeLeafNodeCount * this.activeLeafByteSizeEstimate + this.inactiveLeafNodeCount
00958                 * this.inactiveLeafByteSizeEstimate)
00959                 * this.byteSizeEstimateOverheadFraction > this.maxByteSizeOption.getValue())) {
00960             FoundNode[] learningNodes = findLearningNodes();
00961             Arrays.sort(learningNodes, new Comparator<FoundNode>() {
00962 
00963                 public int compare(FoundNode fn1, FoundNode fn2) {
00964                     if (HoeffdingOptionTree.this.memoryStrategyOption.getValue() == 0) {
00965                         // strategy 1 - every leaf treated equal
00966                         return Double.compare(fn1.node.calculatePromise(),
00967                                 fn2.node.calculatePromise());
00968                     } else if (HoeffdingOptionTree.this.memoryStrategyOption.getValue() == 1) {
00969                         // strategy 2 - internal leaves penalised
00970                         double p1 = fn1.node.calculatePromise();
00971                         if (fn1.parentBranch == -999) {
00972                             p1 /= fn1.parent.getHeadOptionCount();
00973                         }
00974                         double p2 = fn2.node.calculatePromise();
00975                         if (fn2.parentBranch == -999) {
00976                             p1 /= fn2.parent.getHeadOptionCount();
00977                         }
00978                         return Double.compare(p1, p2);
00979                     } else {
00980                         // strategy 3 - all true leaves beat internal leaves
00981                         if (fn1.parentBranch == -999) {
00982                             if (fn2.parentBranch == -999) {
00983                                 return Double.compare(fn1.node.calculatePromise(), fn2.node.calculatePromise());
00984                             }
00985                             return -1; // fn1 < fn2
00986                         }
00987                         if (fn2.parentBranch == -999) {
00988                             return 1; // fn1 > fn2
00989                         }
00990                         return Double.compare(fn1.node.calculatePromise(),
00991                                 fn2.node.calculatePromise());
00992                     }
00993                 }
00994             });
00995             int maxActive = 0;
00996             while (maxActive < learningNodes.length) {
00997                 maxActive++;
00998                 if ((maxActive * this.activeLeafByteSizeEstimate + (learningNodes.length - maxActive)
00999                         * this.inactiveLeafByteSizeEstimate)
01000                         * this.byteSizeEstimateOverheadFraction > this.maxByteSizeOption.getValue()) {
01001                     maxActive--;
01002                     break;
01003                 }
01004             }
01005             int cutoff = learningNodes.length - maxActive;
01006             for (int i = 0; i < cutoff; i++) {
01007                 if (learningNodes[i].node instanceof ActiveLearningNode) {
01008                     deactivateLearningNode(
01009                             (ActiveLearningNode) learningNodes[i].node,
01010                             learningNodes[i].parent,
01011                             learningNodes[i].parentBranch);
01012                 }
01013             }
01014             for (int i = cutoff; i < learningNodes.length; i++) {
01015                 if (learningNodes[i].node instanceof InactiveLearningNode) {
01016                     activateLearningNode(
01017                             (InactiveLearningNode) learningNodes[i].node,
