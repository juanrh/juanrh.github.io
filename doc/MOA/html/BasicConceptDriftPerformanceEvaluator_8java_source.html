00069     public void addResult(Instance inst, double[] classVotes) {
00070         //classVotes[0] -> is Change

00073         //classVotes[3] -> estimation

00076         if (inst.weight() > 0.0 && classVotes.length == 4) {
00077             if (inst.numAttributes() > 1) {
00078                 //if there is ground truth we monitor delay
00079                 this.delay++;

00081             this.weightObserved += inst.weight();
00082             if ( classVotes[0] == 1.0) {
00083                 //Change detected

00085                 this.numberDetections += inst.weight();
00086                 if (this.hasChangeOccurred == true) { 
00087                     this.totalDelay += this.delay - classVotes[2];
00088                     this.numberDetectionsOccurred += inst.weight();
00089                     this.hasChangeOccurred = false;
00090                 }
00091             }
00092             if (this.hasChangeOccurred && classVotes[1] == 1.0) {
00093                 //Warning detected
00094                 //System.out.println("Warning detected at "+getTotalWeightObserved());
00095                 if (this.isWarningZone == false) {
00096                     this.numberWarnings += inst.weight();
00097                     this.isWarningZone = true;
00098                 } 
00099             } else {
00100                 this.isWarningZone = false;
00101             }
00102             if (inst.numAttributes() > 1) {
00103                 if (inst.value(inst.numAttributes() - 2) == 1.0) {//Attribute 1
00104                     //Ground truth Change
00105                     this.numberChanges += inst.weight();
00106                     this.delay = 0;
00107                     this.hasChangeOccurred = true;
00108                 }
00109             }
00110             //Compute error prediction
00111             if (classVotes.length > 1) {
00112                 this.errorPrediction += Math.abs(classVotes[3] - inst.value(0));
