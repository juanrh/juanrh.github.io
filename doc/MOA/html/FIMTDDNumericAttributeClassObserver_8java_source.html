00149         // Return null if the current node is null or we have finished looking through all the possible splits
00150         if (currentNode == null || countRightTotal == 0.0) {

00159         sumTotalRight -= currentNode.leftStatistics.getValue(1);
00160         sumSqTotalLeft += currentNode.leftStatistics.getValue(2);
00161         sumSqTotalRight -= currentNode.leftStatistics.getValue(2);

00163         countRightTotal -= currentNode.leftStatistics.getValue(0);

00167         double merit = criterion.getMeritOfSplit(preSplitDist, postSplitDists);
00168 
00169         if ((currentBestOption == null) || (merit > currentBestOption.merit)) {

00176         if (currentNode.right != null) {
