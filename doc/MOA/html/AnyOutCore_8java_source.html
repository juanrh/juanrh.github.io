00154         double resultDensity = 0;

00156         // exponent = -0.5 (x- mu ) * Sigma ^-1 * (x- mu ) )
00157         //                = -0.5 ( sum_d ((x_i - mu_i)^2 / var_i) )
00158         double exponent = 0.0;
00159         double[] mu = entry.getCenter();
00160 
00161         // factor = 1 / sqrt ( (2*PI)^d * \prod_d(variance) )
00162         //        = 1 / sqrt ((2*PI)^d) * sqrt (\prod_d(variance))
00163         //        = 1 / (2*PI)^(d/2)   * \prod_d( sqrt (variance))
