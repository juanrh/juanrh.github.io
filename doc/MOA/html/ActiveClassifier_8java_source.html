00190 
00191         double costNow;
00192 
00193         if (this.iterationControl <= this.numInstancesInitOption.getValue()) {
00194             costNow = 0;
00195             //Use all instances at the beginning
00196             this.classifier.trainOnInstance(inst);
00197         this.costLabeling++;
00198             return;
00199         } else {
00200             costNow = (this.costLabeling - this.numInstancesInitOption.getValue()) / ((double) this.iterationControl - this.numInstancesInitOption.getValue());
00201         }
00202 
00203 
00204         if (costNow < this.budgetOption.getValue()) { //allow to label
00205             switch (this.activeLearningStrategyOption.getChosenIndex()) {
00206                 case 0: //Random
00207                     labelRandom(inst);
00208                     break;
00209                 case 1: //fixed
00210                     maxPosterior = getMaxPosterior(this.classifier.getVotesForInstance(inst));
00211                     labelFixed(maxPosterior, inst);
00212                     break;
00213                 case 2: //variable
00214                     maxPosterior = getMaxPosterior(this.classifier.getVotesForInstance(inst));
00215                     labelVar(maxPosterior, inst);
00216                     break;
00217                 case 3: //randomized
00218                     maxPosterior = getMaxPosterior(this.classifier.getVotesForInstance(inst));
00219                     maxPosterior = maxPosterior / (this.classifierRandom.nextGaussian() + 1.0);
00220                     labelVar(maxPosterior, inst);
00221                     break;

00223                     maxPosterior = getMaxPosterior(this.classifier.getVotesForInstance(inst));
