00270     public static boolean contextIsCompatible(InstancesHeader originalContext,
00271             InstancesHeader newContext) {
00272 
00273         if (newContext.numClasses() < originalContext.numClasses()) {
00274             return false; // rule 1
00275         }
00276         if (newContext.numAttributes() < originalContext.numAttributes()) {
00277             return false; // rule 2
00278         }
00279         int oPos = 0;
00280         int nPos = 0;
00281         while (oPos < originalContext.numAttributes()) {
00282             if (oPos == originalContext.classIndex()) {
00283                 oPos++;
00284                 if (!(oPos < originalContext.numAttributes())) {
00285                     break;
00286                 }
00287             }
00288             if (nPos == newContext.classIndex()) {
00289                 nPos++;
00290             }
00291             if (originalContext.attribute(oPos).isNominal()) {
00292                 if (!newContext.attribute(nPos).isNominal()) {
00293                     return false; // rule 4
00294                 }
00295                 if (newContext.attribute(nPos).numValues() < originalContext.attribute(oPos).numValues()) {
00296                     return false; // rule 3
00297                 }
00298             } else {
00299                 assert (originalContext.attribute(oPos).isNumeric());
00300                 if (!newContext.attribute(nPos).isNumeric()) {
00301                     return false; // rule 4
00302                 }
00303             }
00304             oPos++;
00305             nPos++;
00306         }
00307         return true; // all checks clear
00308     }
