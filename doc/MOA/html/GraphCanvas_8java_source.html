00148     private boolean updateMinMaxValues() {
00149         double min_y_value_new = min_y_value;
00150         double max_y_value_new = max_y_value;
00151         double max_x_value_new = max_x_value;
00152 
00153         if (measure0 != null && measure1 != null) {
00154             min_y_value_new = Math.min(measure0.getMinValue(measureSelected), measure1.getMinValue(measureSelected));
00155             max_y_value_new = Math.max(measure0.getMaxValue(measureSelected), measure1.getMaxValue(measureSelected));
00156             max_x_value_new = Math.max(measure0.getNumberOfValues(measureSelected), max_x_value);
00157         } else {
00158             if (measure0 != null) {
00159                 min_y_value_new = measure0.getMinValue(measureSelected);
00160                 max_y_value_new = measure0.getMaxValue(measureSelected);
00161                 max_x_value_new = Math.max(measure0.getNumberOfValues(measureSelected), max_x_value);
00162             }
00163         }
00164 
00165         //resizing needed?
00166         if (max_x_value_new != max_x_value || max_y_value_new != max_y_value || min_y_value_new != min_y_value) {
00167             min_y_value = min_y_value_new;
00168             max_y_value = max_y_value_new;
00169             max_x_value = max_x_value_new;
00170             return true;
00171         }
00172         return false;
00173     }
