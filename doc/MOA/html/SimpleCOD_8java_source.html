00145                 int count = x.count_after + x.CountPrecNeighs(GetWindowStart());
00146                 if (count < m_k) {
00147                     if (bTrace) Println("x is an outlier");

00149                     SaveOutlier(x);
00150                 } else {
00151                     if (bTrace) Println("x is an inlier, add to event queue");
