00578     List[]  result;
00579     Patch   patch;
00580     int     from;
00581     int     to;
00582     int     sizeDiff;
00583 
00584     result                       = new List[SIDEBYSIDE_SIZE];
00585     result[SIDEBYSIDE_FIRST]     = new ArrayList();
00586     result[SIDEBYSIDE_SECOND]    = new ArrayList();
00587     result[SIDEBYSIDE_INDICATOR] = new ArrayList();
00588     result[SIDEBYSIDE_STARTPOS]  = new ArrayList();
00589     result[SIDEBYSIDE_ENDPOS]    = new ArrayList();
00590     patch                        = difflib.DiffUtils.diff(list1, list2);
00591 
00592     // the same?
00593     if (patch.getDeltas().size() == 0) {
00594       result[SIDEBYSIDE_FIRST].addAll(list1);
00595       result[SIDEBYSIDE_SECOND].addAll(list2);
00596       addToList(0, list1.size() - 1, INDICATOR_SAME, result[SIDEBYSIDE_INDICATOR]);
00597       return new SideBySideDiff(result);
00598     }
00599 
00600     to = 0;
00601     for (Delta delta: patch.getDeltas()) {
00602       from = to;
00603 
00604       // common content
00605       to = delta.getOriginal().getPosition() - 1;
00606       addToList(from, to, list1,          result[SIDEBYSIDE_FIRST]);
00607       addToList(from, to, list1,          result[SIDEBYSIDE_SECOND]);
00608       addToList(from, to, INDICATOR_SAME, result[SIDEBYSIDE_INDICATOR]);
00609 
00610       // start pos of delta
00611       result[SIDEBYSIDE_STARTPOS].add(result[SIDEBYSIDE_FIRST].size());
00612 
00613       if (delta instanceof InsertDelta) {
00614     // added content
00615     addToList(0, delta.getRevised().size() - 1, new Filler(), result[SIDEBYSIDE_FIRST]);
00616     result[SIDEBYSIDE_SECOND].addAll(delta.getRevised().getLines());
00617     addToList(0, delta.getRevised().size() - 1, INDICATOR_ADDED, result[SIDEBYSIDE_INDICATOR]);
00618     to = delta.getOriginal().getPosition() + delta.getOriginal().size();
00619       }
00620       else if (delta instanceof DeleteDelta) {
00621     // deleted content
00622     result[SIDEBYSIDE_FIRST].addAll(delta.getOriginal().getLines());
00623     addToList(0, delta.getOriginal().size() - 1, new Filler(), result[SIDEBYSIDE_SECOND]);
00624     addToList(0, delta.getOriginal().size() - 1, INDICATOR_DELETED, result[SIDEBYSIDE_INDICATOR]);
00625     to = delta.getOriginal().getPosition() + delta.getOriginal().size();
00626       }
00627       else {
00628     // changed content
00629     result[SIDEBYSIDE_FIRST].addAll(delta.getOriginal().getLines());
00630     result[SIDEBYSIDE_SECOND].addAll(delta.getRevised().getLines());
00631     addToList(1, Math.max(delta.getOriginal().size(), delta.getRevised().size()), INDICATOR_CHANGED, result[SIDEBYSIDE_INDICATOR]);
00632     // filler necessary?
00633     sizeDiff = delta.getRevised().size() - delta.getOriginal().size();
00634     if (sizeDiff > 0)
00635       addToList(1, sizeDiff, new Filler(), result[SIDEBYSIDE_FIRST]);
00636     if (sizeDiff < 0)
00637       addToList(1, -sizeDiff, new Filler(), result[SIDEBYSIDE_SECOND]);
00638     to = delta.getOriginal().getPosition() + delta.getOriginal().size();
00639       }
00640 
00641       // end pos of delta
00642       result[SIDEBYSIDE_ENDPOS].add(result[SIDEBYSIDE_FIRST].size());
00643     }
00644 
00645     // trailing common content?
00646     if (to < list1.size()) {
00647       addToList(to, list1.size() - 1, list1,          result[SIDEBYSIDE_FIRST]);
00648       addToList(to, list1.size() - 1, list1,          result[SIDEBYSIDE_SECOND]);
00649       addToList(to, list1.size() - 1, INDICATOR_SAME, result[SIDEBYSIDE_INDICATOR]);
00650     }
