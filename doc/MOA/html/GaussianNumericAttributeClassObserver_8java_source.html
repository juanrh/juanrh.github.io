00108     public double[] getSplitPointSuggestions() {
00109         Set<Double> suggestedSplitValues = new TreeSet<Double>();
00110         double minValue = Double.POSITIVE_INFINITY;
00111         double maxValue = Double.NEGATIVE_INFINITY;
00112         for (int i = 0; i < this.attValDistPerClass.size(); i++) {
00113             GaussianEstimator estimator = this.attValDistPerClass.get(i);
00114             if (estimator != null) {
00115                 if (this.minValueObservedPerClass.getValue(i) < minValue) {
00116                     minValue = this.minValueObservedPerClass.getValue(i);
00117                 }
00118                 if (this.maxValueObservedPerClass.getValue(i) > maxValue) {
00119                     maxValue = this.maxValueObservedPerClass.getValue(i);
00120                 }
00121             }
00122         }
00123         if (minValue < Double.POSITIVE_INFINITY) {
00124             double range = maxValue - minValue;
00125             for (int i = 0; i < this.numBinsOption.getValue(); i++) {
00126                 double splitValue = range / (this.numBinsOption.getValue() + 1.0) * (i + 1)
00127                         + minValue;
00128                 if ((splitValue > minValue) && (splitValue < maxValue)) {
00129                     suggestedSplitValues.add(splitValue);
00130                 }
00131             }
00132         }
00133         double[] suggestions = new double[suggestedSplitValues.size()];
00134         int i = 0;
00135         for (double suggestion : suggestedSplitValues) {
00136             suggestions[i++] = suggestion;
00137         }
