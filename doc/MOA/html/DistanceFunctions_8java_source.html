00123     public static double euclidean(EuclideanCoordinate coord1, EuclideanCoordinate coord2) {
00124         int size = Math.min(coord1.dimensions(), coord2.dimensions());
00125         double distance = 0;
00126         for(int i = 0; i < size; i++) {
00127             double diff = coord1.get(i) - coord2.get(i);
00128             distance += diff * diff;
00129         }
00130         distance = Math.sqrt(distance);
00131         return distance;
