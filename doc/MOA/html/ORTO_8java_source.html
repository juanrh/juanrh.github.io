01087         public void trainOnInstanceImpl(Instance inst, ORTO ft) {
01088             // Initialize Perceptron if necessary   
01089             if (this.reset == true) {
01090                 this.reset = false;
01091                 this.weightAttribute = new double[inst.numAttributes()];
01092                 this.instancesSeen = 0;
01093                 this.attributeStatistics = new DoubleVector();
01094                 this.squaredAttributeStatistics = new DoubleVector();
01095                 for (int j = 0; j < inst.numAttributes(); j++) {
01096                     weightAttribute[j] = 2 * ft.classifierRandom.nextDouble() - 1;
01097                 }
01098             }
01099             
01100             // Update attribute statistics
01101             instancesSeen++;
01102             for(int j = 0; j < inst.numAttributes() -1; j++)
01103             {
01104                 attributeStatistics.addToValue(j, inst.value(j));    
01105                 squaredAttributeStatistics.addToValue(j, inst.value(j)*inst.value(j));
01106             }
01107             
01108             // Update weights
01109             
01110             double learningRatio = 0.0;
01111             if(ft.LearningRatioDecayOrConstOption.isSet()){
01112                 learningRatio = ft.LearningRatioOption.getValue();
01113             } else {
01114                 learningRatio = ft.initLearnRate / (1 + instancesSeen * ft.learnRateDecay);
01115             }
01116             // double learningRatio = ft.learningRatioOption.getValue();
01117             double actualClass = inst.classValue();
01118             double predictedClass = this.prediction(inst);
01119             
01120             // SET DELTA TO ACTUAL - PREDICTED, NOT PREDICTED - ACTUAL AS SAID IN PAPER
01121             double delta = actualClass - predictedClass;
01122             
01123             for (int j = 0; j < inst.numAttributes() - 1; j++) {
01124                 
01125                 if (inst.attribute(j).isNumeric()) {
01126                     // Update weights. Ensure attribute values are normalized first
01127                     double sd = Math.sqrt((squaredAttributeStatistics.getValue(j) - ((attributeStatistics.getValue(j) * attributeStatistics.getValue(j))/instancesSeen))/instancesSeen);
01128                     double instanceValue = 0;
01129                     if (sd > 0.0000001) { // Limit found in implementation by Ikonomovska et al (2011)
01130                         instanceValue = (inst.value(j) - (attributeStatistics.getValue(j)/instancesSeen))/(3*sd);
01131                     }
01132                     this.weightAttribute[j] += learningRatio * delta * instanceValue;
01133                 }
01134             }
01135             this.weightAttribute[inst.numAttributes() - 1] += learningRatio * delta;
01136         }
