00133         double weight = inst.weight();
00134         int trueClass = (int) inst.classValue();
00135         if (weight > 0.0) {
00136             if (TotalweightObserved == 0) {

00139             this.TotalweightObserved += weight;
00140             this.weightObserved.add(weight);

00142             if (predictedClass == trueClass) {
00143                 this.weightCorrect.add(weight);
00144             } else {
00145                 this.weightCorrect.add(0);
00146             }
00147             //Add Kappa statistic information
00148             for (int i = 0; i < this.numClasses; i++) {
00149                 this.rowKappa[i].add(i == predictedClass ? weight : 0);
00150                 this.columnKappa[i].add(i == trueClass ? weight : 0);
00151             }
00152             if (this.lastSeenClass == trueClass) {
00153                 this.weightCorrectNoChangeClassifier.add(weight);
00154             } else {
00155                 this.weightCorrectNoChangeClassifier.add(0);
00156             }
00157             this.classAccuracy[trueClass].add(predictedClass == trueClass ? weight : 0.0);
00158             this.lastSeenClass = trueClass;

00186     public double getKappaStatistic() {
00187         if (this.weightObserved.total() > 0.0) {
00188             double p0 = this.weightCorrect.total() / this.weightObserved.total();
00189             double pc = 0;
00190             for (int i = 0; i < this.numClasses; i++) {
00191                 pc += (this.rowKappa[i].total() / this.weightObserved.total())
00192                         * (this.columnKappa[i].total() / this.weightObserved.total());
00193             }
00194             return (p0 - pc) / (1 - pc);

00196             return 0;
