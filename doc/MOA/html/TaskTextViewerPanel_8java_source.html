00327     public void setGraph(String preview) {
00328         //Change the graph when there is change in the text
00329         double processFrequency = 1000;
00330         if (preview != null && !preview.equals("")) {
00331             MeasureCollection oldAccuracy = acc1[0];
00332             acc1[0] = getNewMeasureCollection();
00333             Scanner scanner = new Scanner(preview);
00334             String firstLine = scanner.nextLine();
00335             boolean isSecondLine = true;
00336 
00337             boolean isPrequential = firstLine.startsWith("learning evaluation instances,evaluation time");
00338             boolean isHoldOut = firstLine.startsWith("evaluation instances,to");
00339             int accuracyColumn = 6;
00340             int kappaColumn = 4;
00341             int RamColumn = 2;
00342             int timeColumn = 1;
00343             int memoryColumn = 9;
00344             int kappaTempColumn = 5;
00345             if (this.taskManagerPanel instanceof CDTaskManagerPanel) {
00346                 accuracyColumn = 6;
00347                 kappaColumn = 4;
00348                 RamColumn = 2;
00349                 timeColumn = 1;
00350                 memoryColumn = 9;
00351             } else if (isPrequential || isHoldOut) {
00352                 accuracyColumn = 4;
00353                 kappaColumn = 5;
00354                 RamColumn = 2;
00355                 timeColumn = 1;
00356                 memoryColumn = 7;
00357                 kappaTempColumn = 5;
00358                 String[] tokensFirstLine = firstLine.split(",");
00359                 int i = 0;
00360                 for (String s : tokensFirstLine) {
00361 
00362                     if (s.equals("classifications correct (percent)")) {
00363                         accuracyColumn = i;
00364                     } else if (s.equals("Kappa Statistic (percent)")) {
00365                         kappaColumn = i;
00366                     } else if (s.equals("Kappa Temporal Statistic (percent)")) {
00367                         kappaTempColumn = i;
00368                     } else if (s.equals("model cost (RAM-Hours)")) {
00369                         RamColumn = i;
00370                     } else if (s.equals("evaluation time (cpu seconds)")
00371                             || s.equals("total train time")) {
00372                         timeColumn = i;
00373                     } else if (s.equals("model serialized size (bytes)")) {
00374                         memoryColumn = i;
00375                     }
00376                     i++;
00377                 }
00378             }
00379             if (isPrequential || isHoldOut || this.taskManagerPanel instanceof CDTaskManagerPanel) {
00380                 while (scanner.hasNextLine()) {
00381                     String line = scanner.nextLine();
00382                     String[] tokens = line.split(",");
00383                     this.acc1[0].addValue(0, round(parseDouble(tokens[accuracyColumn])));
00384                     this.acc1[0].addValue(1, round(parseDouble(tokens[kappaColumn])));
00385                     this.acc1[0].addValue(2, round(parseDouble(tokens[kappaTempColumn])));
00386                     if (!isHoldOut) {
00387                         this.acc1[0].addValue(3, Math.abs(parseDouble(tokens[RamColumn])));
00388                     }
00389                     this.acc1[0].addValue(4, round(parseDouble(tokens[timeColumn])));
00390                     this.acc1[0].addValue(5, round(parseDouble(tokens[memoryColumn]) / (1024 * 1024)));
00391 
00392                     if (isSecondLine == true) {
00393                         processFrequency = Math.abs(parseDouble(tokens[0]));
00394                         isSecondLine = false;
00395                         if (acc1[0].getValue(0, 0) != oldAccuracy.getValue(0, 0)) { //(!line.equals(secondLine)) {
00396                             //If we are in a new task, compare with the previous
00397                             secondLine = line;
00398                             if (processFrequency == this.graphCanvas.getProcessFrequency()) {
00399                                 acc2[0] = oldAccuracy;
00400                             }
00401                         }
00402                     }
00403                 }
00404             } else {
00405                 this.acc2[0] = getNewMeasureCollection();
00406             }
00407 
00408         } else {
00409             this.acc1[0] = getNewMeasureCollection();
00410             this.acc2[0] = getNewMeasureCollection();
00411         }
00412 
00413
