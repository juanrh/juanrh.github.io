00371 
00372         int L = P.length;
00373 
00374         double Q[][] = new double[L][L];
00375 
00376         // set priors
00377         for (int j = 0; j < L; j++) {
00378             Q[j][j] = P[j];
00379         }
00380 
00381         // create conditionals
00382         for (int j = 0; j < Q.length; j++) {
00383             for (int k = j + 1; k < Q[j].length; k++) {
00384 
00385                 // dependence
00386                 if (M[j][k]) {
00387                     // min = tending toward mutual exclusivity
00388                     // max = tending toward total co-occurence
00389                     // @NOTE it would also be an option to select in [min,max], but since
00390                     //       we are approximating the joint distribution, we can take
00391                     //       a stronger approach, and just take either min or max.
00392                     Q[j][k] = (m_MetaRandom.nextBoolean() ? min(P[j], P[k]) : max(P[j], P[k]));
00393                     Q[k][j] = (Q[j][k] * Q[j][j]) / Q[k][k]; // Bayes Rule
00394                 } // independence
00395                 else {
00396                     Q[j][k] = P[j];
00397                     Q[k][j] = (Q[j][k] * P[k]) / P[j]; // Bayes Rule
00398                 }
00399             }
00400         }
00401 
00402         return Q;
